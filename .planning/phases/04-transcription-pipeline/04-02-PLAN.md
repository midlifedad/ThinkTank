       ```python
           else:

       ```

    f. **Pass 2 -- Existing transcript** (if source has transcript_url_pattern):
       ```python
       if transcript is None:
           pattern = source.config.get("transcript_url_pattern")
           if pattern:

               if transcript:

       ```

       ```python
       if transcript is None:
           import tempfile, os

               return await transcribe_with_chunking(wav_path, content.duration_seconds)

           transcript = await transcribe_via_gpu(content.url, tmp_dir, _gpu_fn)
           if transcript:
               method = "parakeet"
       ```
